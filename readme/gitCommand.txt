git init  - включение, запуск нового репозитория
git reset  - отмена изменений до предыдущего коммита
git log  - просмотреть лог комиттов
git branch <branch-name> - создать новую ветку
git branch -d <branch-name> - удалить локальную ветку
git branch -r <branch-name> - удалить удаленную ветку
git branch --show-current -посмотреть текущую ветку
git status - посмотреть статус текущей ветки
git branch or git branch --list - просмотреть список веток
git checkout <name-of-your-branch> - перейти на другую ветку
git checkout -b <name branch> -переключится на другую ветку
git add <file> - дабавить файл в индекс для дальнейшего комитта
git add . - дабавить содержимое папки в индекс для дальнейшего комитта
git commit -m "commit message" - сделать комит изменений проекта с заметкой
git push <remote> <branch-name> - отправить проект на сервер
git remote add origin https://gitlab.com/cherniavskyi/my-list.git - добавить на сервер
git pull <remote> <branch-name> - сохранить изменения сделание на сервере
git clone <https://name-of-the-repository-link> <name package> - викачать проект с сервера
git cherry-pick <commit-hash> - копировать комит в другую ветку
